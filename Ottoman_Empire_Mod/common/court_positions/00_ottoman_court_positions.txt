# Ottoman Court Positions

grand_vizier_ottoman = {
	skill = stewardship
	max_available_positions = 1
	
	opinion = {
		same_culture_opinion = 10
		different_culture_opinion = 5
	}
	
	valid_position = {
		is_adult = yes
		is_imprisoned = no
		NOT = { has_trait = incapable }
		trigger_if = {
			limit = { exists = liege }
			NOT = { 
				is_at_war_with = liege 
			}
		}
	}
	
	is_shown = {
		scope:liege = {
			culture = ottoman
			has_government = ottoman_sultanate_government
		}
	}
	
	valid_character = {
		scope:employee = {
			is_courtier_of = scope:liege  
			stewardship >= 12
			is_adult = yes
			NOT = { has_trait = incapable }
		}
	}
	
	revoke_cost = {
		prestige = 25
	}
	
	salary = {
		gold = 3
	}
	
	is_powerful_vassal = yes
	
	employer_modifier = {
		domain_limit = 2
		vassal_opinion = 10
		monthly_county_control_growth_factor = 0.2
		monthly_prestige = 0.3
	}
	
	employee_modifier = {
		monthly_prestige = 0.5
		stewardship = 3
		diplomacy = 1
	}
	
	candidate_score = {
		value = 100
		add = scope:employee.stewardship
		
		if = {
			limit = { scope:employee = { has_trait = administrator } }
			add = 25
		}
		
		if = {
			limit = { scope:employee = { culture = scope:liege.culture } }
			add = 15
		}
		
		if = {
			limit = { scope:employee = { has_trait = grand_vizier } }
			add = 50
		}
	}
}
