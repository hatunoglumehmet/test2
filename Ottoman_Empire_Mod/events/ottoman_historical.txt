# Ottoman Historical Events

namespace = ottoman_historical

# The Rise of the Ottoman Beylik
ottoman_historical.001 = {
	type = character_event
	title = ottoman_historical.001.t
	desc = ottoman_historical.001.desc
	theme = realm
	
	left_portrait = {
		character = root
		animation = personality_bold
	}
	
	trigger = {
		is_ruler = yes
		culture = ottoman
		current_date >= 1299.1.1
		current_date <= 1350.1.1
		highest_held_title_tier = tier_duchy
		NOT = { has_character_flag = ottoman_rise_event }
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 3
			has_trait = ambitious
		}
	}
	
	immediate = {
		add_character_flag = ottoman_rise_event
	}
	
	option = {
		name = ottoman_historical.001.a
		
		add_trait = ghazi_warrior
		add_prestige = 500
		add_piety = 300
		
		every_neighboring_realm_same_rank_owner = {
			limit = { 
				NOT = { religion = root.religion }
			}
			add_opinion = {
				target = root
				modifier = rising_power
				opinion = -20
			}
		}
	}
}

# The Devshirme System Implementation
ottoman_historical.004 = {
	type = character_event
	title = ottoman_historical.004.t
	desc = ottoman_historical.004.desc
	theme = realm
	
	left_portrait = {
		character = root
		animation = personality_rational
	}
	
	trigger = {
		is_ruler = yes
		culture = ottoman
		current_date >= 1380.1.1
		current_date <= 1600.1.1
		NOT = { has_character_flag = devshirme_system_event }
		culture = { has_innovation = innovation_devshirme_system }
	}
	
	immediate = {
		add_character_flag = devshirme_system_event
	}
	
	option = {
		name = ottoman_historical.004.a
		
		add_prestige = 300
		add_piety = -50
		
		every_realm_county = {
			limit = { NOT = { culture = root.culture } }
			add_county_modifier = {
				modifier = devshirme_recruitment
				years = 50
			}
		}
		
		create_character = {
			location = capital_province
			template = devshirme_officer_template
			culture = ottoman
			faith = sunni
			dynasty = none
			save_scope_as = new_devshirme
		}
		
		add_courtier = scope:new_devshirme
	}
	
	option = {
		name = ottoman_historical.004.b
		
		add_piety = 100
		add_stress = 50
		
		different_culture_opinion = 20
	}
}

# The Ottoman-Byzantine War
ottoman_historical.005 = {
	type = character_event
	title = ottoman_historical.005.t
	desc = ottoman_historical.005.desc
	theme = war
	
	left_portrait = {
		character = root
		animation = personality_bold
	}
	
	trigger = {
		is_ruler = yes
		culture = ottoman
		current_date >= 1450.1.1
		current_date <= 1460.1.1
		exists = title:e_byzantine_empire.holder
		NOT = { has_character_flag = byzantine_war_event }
		NOT = { is_at_war_with = title:e_byzantine_empire.holder }
	}
	
	immediate = {
		add_character_flag = byzantine_war_event
	}
	
	option = {
		name = ottoman_historical.005.a
		
		title:e_byzantine_empire.holder = {
			save_scope_as = byzantine_emperor
		}
		
		start_war = {
			cb = conquest_cb
			target = scope:byzantine_emperor
			target_title = title:c_byzantion
		}
		
		add_prestige = 200
		add_piety = 300
		
		trigger_event = {
			id = conquest_constantinople.001
			days = 60
		}
	}
	
	option = {
		name = ottoman_historical.005.b
		
		add_stress = 75
		add_trait = craven
		
		same_faith_opinion = -20
	}
}
