# Ottoman Empire Decisions

form_ottoman_empire_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_realm.dds"
	
	desc = form_ottoman_empire_decision_desc
	selection_tooltip = form_ottoman_empire_decision_tooltip
	
	is_shown = {
		is_ruler = yes
		culture = ottoman
		highest_held_title_tier = tier_kingdom
		NOT = { has_title = title:e_ottoman }
		religion = { is_in_family = rf_islamic }
	}
	
	is_valid_showing_failures_only = {
		is_independent_ruler = yes
		prestige_level >= 3
		completely_controls = title:c_byzantion
		OR = {
			completely_controls = title:d_thrace
			completely_controls = title:d_anatolia
		}
		gold >= 500
	}
	
	is_valid = {
		trigger_if = {
			limit = { exists = title:e_ottoman.holder }
			NOT = { title:e_ottoman.holder = root }
		}
	}
	
	cost = {
		gold = 500
		prestige = 1000
	}
	
	effect = {
		create_title_and_vassal_change = {
			type = created
			save_scope_as = change
			add_claim_on_loss = no
		}
		
		title:e_ottoman = {
			change_title_holder = {
				holder = root
				change = scope:change
			}
		}
		
		resolve_title_and_vassal_change = scope:change
		
		add_trait = sultan_of_sultans
		add_prestige = 2000
		add_piety = 500
		
		every_vassal = {
			limit = { religion = root.religion }
			add_opinion = {
				target = root
				modifier = formed_great_empire
				opinion = 50
			}
		}
		
		capital_province = {
			add_province_modifier = {
				modifier = imperial_capital
				years = 100
			}
		}
	}
	
	ai_potential = {
		always = yes
	}
	
	ai_will_do = {
		base = 100
	}
}

conquer_constantinople_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_major_war.dds"
	
	desc = conquer_constantinople_decision_desc
	selection_tooltip = conquer_constantinople_decision_tooltip
	
	is_shown = {
		is_ruler = yes
		culture = ottoman
		NOT = { has_trait = conqueror_of_constantinople }
		exists = title:c_byzantion.holder
		NOT = { title:c_byzantion.holder = root }
		title:c_byzantion.holder = {
			OR = {
				culture = greek
				culture = roman
			}
		}
	}
	
	is_valid_showing_failures_only = {
		martial >= 15
		gold >= 1000
		prestige_level >= 2
		has_trait = ghazi_warrior
		army_size >= 15000
	}
	
	cost = {
		gold = 1000
		prestige = 500
		piety = 200
	}
	
	effect = {
		title:c_byzantion.holder.top_liege = {
			save_scope_as = byzantine_emperor
		}
		
		start_war = {
			cb = conquest_cb
			target = scope:byzantine_emperor
			target_title = title:c_byzantion
		}
		
		add_character_modifier = {
			modifier = preparing_great_conquest
			years = 5
		}
		
		trigger_event = {
			id = conquest_constantinople.001
			days = 30
		}
	}
	
	ai_potential = {
		has_trait = ambitious
		martial >= 12
	}
	
	ai_will_do = {
		base = 50
		modifier = {
			add = 50
			has_trait = brave
		}
		modifier = {
			add = 25
			has_trait = zealous
		}
	}
}
