# Ottoman Empire General Events

namespace = ottoman_empire

# Ottoman Expansion Event
ottoman_empire.001 = {
	type = character_event
	title = ottoman_empire.001.t
	desc = ottoman_empire.001.desc
	theme = realm
	
	left_portrait = {
		character = root
		animation = personality_bold
	}
	
	trigger = {
		is_ruler = yes
		culture = ottoman
		has_trait = ottoman_legacy
		realm_size >= 15
		NOT = { has_character_flag = ottoman_expansion_event }
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 2
			has_trait = ambitious
		}
	}
	
	immediate = {
		add_character_flag = ottoman_expansion_event
	}
	
	option = {
		name = ottoman_empire.001.a
		
		add_prestige = 300
		add_trait = administrator
		
		every_vassal = {
			add_opinion = {
				target = root
				modifier = efficient_administrator
				opinion = 15
			}
		}
	}
	
	option = {
		name = ottoman_empire.001.b
		
		add_gold = 500
		add_trait = greedy
		
		every_vassal = {
			add_opinion = {
				target = root
				modifier = heavy_taxation
				opinion = -10
			}
		}
	}
}

# Devshirme System Event
ottoman_empire.004 = {
	type = character_event
	title = ottoman_empire.004.t
	desc = ottoman_empire.004.desc
	theme = realm
	
	left_portrait = {
		character = root
		animation = personality_rational
	}
	
	trigger = {
		is_ruler = yes
		culture = ottoman
		culture = { has_innovation = innovation_devshirme_system }
		any_realm_county = {
			NOT = { culture = root.culture }
			development >= 10
		}
		NOT = { has_character_flag = devshirme_event }
	}
	
	immediate = {
		add_character_flag = devshirme_event
		random_realm_county = {
			limit = {
				NOT = { culture = root.culture }
				development >= 10
			}
			save_scope_as = devshirme_county
		}
	}
	
	option = {
		name = ottoman_empire.004.a
		
		add_prestige = 150
		
		scope:devshirme_county = {
			add_county_modifier = {
				modifier = devshirme_recruitment
				years = 20
			}
		}
		
		create_character = {
			location = scope:devshirme_county
			template = devshirme_officer_template
			culture = ottoman
			faith = sunni
			dynasty = none
			save_scope_as = new_officer
		}
		
		add_courtier = scope:new_officer
	}
	
	option = {
		name = ottoman_empire.004.b
		
		add_piety = 100
		add_stress = 25
		
		scope:devshirme_county = {
			county_opinion_add = 10
		}
	}
}
