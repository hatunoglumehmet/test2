# Ottoman Cultural and Feast Events

namespace = ottoman_culture

# Ottoman Feast Event
ottoman_culture.001 = {
	type = character_event
	title = ottoman_culture.001.t
	desc = ottoman_culture.001.desc
	theme = feast
	
	left_portrait = {
		character = root
		animation = personality_honorable
	}
	
	right_portrait = {
		character = scope:feast_guest
		animation = personality_content
	}
	
	trigger = {
		is_ruler = yes
		culture = ottoman
		gold >= 300
		NOT = { has_character_flag = ottoman_feast_event }
		any_courtier = {
			is_adult = yes
			NOT = { has_trait = reclusive }
			count >= 5
		}
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 2
			has_trait = gregarious
		}
		modifier = {
			add = 1
			stewardship >= 15
		}
	}
	
	immediate = {
		add_character_flag = ottoman_feast_event
		random_courtier = {
			limit = {
				is_adult = yes
				NOT = { has_trait = reclusive }
			}
			save_scope_as = feast_guest
		}
	}
	
	option = {
		name = ottoman_culture.001.a
		
		add_gold = -300
		add_prestige = 200
		add_stress = -50
		
		court_opinion = 25
		vassal_opinion = 15
		
		random_list = {
			25 = { add_trait = master_calligrapher }
			25 = { add_trait = palace_musician }
			25 = { add_trait = turkish_poet }
			25 = { nothing = yes }
		}
	}
	
	option = {
		name = ottoman_culture.001.b
		
		add_gold = -150
		add_prestige = 100
		
		court_opinion = 10
	}
	
	option = {
		name = ottoman_culture.001.c
		
		add_trait = greedy
		add_stress = 25
		
		court_opinion = -15
	}
}

# Ottoman Music and Poetry
ottoman_culture.006 = {
	type = character_event
	title = ottoman_culture.006.t
	desc = ottoman_culture.006.desc
	theme = court
	
	left_portrait = {
		character = root
		animation = personality_content
	}
	
	trigger = {
		is_ruler = yes
		culture = ottoman
		diplomacy >= 12
		NOT = { has_character_flag = music_poetry_event }
		any_courtier = {
			OR = {
				has_trait = poet
				has_trait = palace_musician
				has_trait = turkish_poet
			}
		}
	}
	
	immediate = {
		add_character_flag = music_poetry_event
		random_courtier = {
			limit = {
				OR = {
					has_trait = poet
					has_trait = palace_musician
					has_trait = turkish_poet
				}
			}
			save_scope_as = artist_courtier
		}
	}
	
	option = {
		name = ottoman_culture.006.a
		
		add_trait = turkish_poet
		add_diplomacy_lifestyle_xp = 150
		add_prestige = 200
		
		court_opinion = 20
		
		scope:artist_courtier = {
			add_opinion = {
				target = root
				modifier = artistic_patron
				opinion = 30
			}
		}
	}
	
	option = {
		name = ottoman_culture.006.b
		
		add_trait = palace_musician
		add_stress = -75
		add_prestige = 100
		
		court_opinion = 25
	}
}
