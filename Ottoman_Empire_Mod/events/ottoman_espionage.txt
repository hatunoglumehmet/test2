# Ottoman Espionage and Diplomatic Events

namespace = ottoman_espionage

# The Spy Network
ottoman_espionage.001 = {
	type = character_event
	title = ottoman_espionage.001.t
	desc = ottoman_espionage.001.desc
	theme = intrigue
	
	left_portrait = {
		character = root
		animation = personality_callous
	}
	
	trigger = {
		is_ruler = yes
		culture = ottoman
		intrigue >= 12
		gold >= 200
		NOT = { has_character_flag = spy_network_event }
		any_neighboring_ruler = {
			NOT = { religion = root.religion }
		}
	}
	
	immediate = {
		add_character_flag = spy_network_event
		random_neighboring_ruler = {
			limit = { NOT = { religion = root.religion } }
			save_scope_as = target_ruler
		}
	}
	
	option = {
		name = ottoman_espionage.001.a
		
		add_gold = -200
		add_intrigue_lifestyle_xp = 150
		
		add_character_modifier = {
			modifier = extensive_spy_network
			years = 10
		}
		
		scope:target_ruler = {
			add_opinion = {
				target = root
				modifier = suspicious_activities
				opinion = -15
			}
		}
	}
	
	option = {
		name = ottoman_espionage.001.b
		
		add_trait = honest
		add_piety = 100
		
		same_faith_opinion = 10
	}
}

# Diplomatic Marriage Proposal
ottoman_espionage.002 = {
	type = character_event
	title = ottoman_espionage.002.t
	desc = ottoman_espionage.002.desc
	theme = diplomacy
	
	left_portrait = {
		character = root
		animation = personality_rational
	}
	
	right_portrait = {
		character = scope:marriage_candidate
		animation = personality_content
	}
	
	trigger = {
		is_ruler = yes
		culture = ottoman
		diplomacy >= 14
		NOT = { has_character_flag = diplomatic_marriage_event }
		any_child = {
			is_adult = no
			age >= 12
			is_married = no
		}
		any_neighboring_ruler = {
			any_child = {
				is_adult = no
				age >= 12
				is_married = no
				NOT = { culture = root.culture }
			}
		}
	}
	
	immediate = {
		add_character_flag = diplomatic_marriage_event
		random_neighboring_ruler = {
			limit = {
				any_child = {
					is_adult = no
					age >= 12
					is_married = no
					NOT = { culture = root.culture }
				}
			}
			random_child = {
				limit = {
					is_adult = no
					age >= 12
					is_married = no
				}
				save_scope_as = marriage_candidate
			}
			save_scope_as = foreign_ruler
		}
	}
	
	option = {
		name = ottoman_espionage.002.a
		
		add_prestige = 200
		add_gold = -300
		
		scope:foreign_ruler = {
			add_opinion = {
				target = root
				modifier = diplomatic_alliance
				opinion = 40
			}
		}
		
		add_character_modifier = {
			modifier = diplomatic_marriage_arranged
			years = 15
		}
	}
	
	option = {
		name = ottoman_espionage.002.b
		
		add_trait = isolationist
		add_prestige = -50
		
		scope:foreign_ruler = {
			add_opinion = {
				target = root
				modifier = rejected_proposal
				opinion = -20
			}
		}
	}
}

# Trade Agreement Negotiation
ottoman_espionage.005 = {
	type = character_event
	title = ottoman_espionage.005.t
	desc = ottoman_espionage.005.desc
	theme = stewardship
	
	left_portrait = {
		character = root
		animation = personality_rational
	}
	
	trigger = {
		is_ruler = yes
		culture = ottoman
		stewardship >= 12
		NOT = { has_character_flag = trade_agreement_event }
		any_realm_county = {
			has_building = grand_bazaar_01
		}
		any_neighboring_ruler = {
			any_realm_county = {
				is_coastal_county = yes
			}
		}
	}
	
	immediate = {
		add_character_flag = trade_agreement_event
		random_neighboring_ruler = {
			limit = {
				any_realm_county = {
					is_coastal_county = yes
				}
			}
			save_scope_as = trading_partner
		}
	}
	
	option = {
		name = ottoman_espionage.005.a
		
		add_gold = 300
		add_prestige = 150
		
		add_character_modifier = {
			modifier = lucrative_trade_deal
			years = 20
		}
		
		scope:trading_partner = {
			add_opinion = {
				target = root
				modifier = profitable_partnership
				opinion = 25
			}
		}
	}
	
	option = {
		name = ottoman_espionage.005.b
		
		add_trait = generous
		add_piety = 100
		
		scope:trading_partner = {
			add_opinion = {
				target = root
				modifier = fair_dealer
				opinion = 30
			}
		}
		
		different_culture_opinion = 15
	}
}
