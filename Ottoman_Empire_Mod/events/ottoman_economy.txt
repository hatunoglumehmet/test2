# Ottoman Economic and Administrative Events

namespace = ottoman_economy

# Tax Collection Reform Event
ottoman_economy.001 = {
	type = character_event
	title = ottoman_economy.001.t
	desc = ottoman_economy.001.desc
	theme = stewardship
	
	left_portrait = {
		character = root
		animation = personality_rational
	}
	
	trigger = {
		is_ruler = yes
		culture = ottoman
		stewardship >= 14
		realm_size >= 20
		NOT = { has_character_flag = tax_reform_event }
		gold >= 300
	}
	
	immediate = {
		add_character_flag = tax_reform_event
	}
	
	option = {
		name = ottoman_economy.001.a
		
		add_gold = -300
		add_prestige = 200
		
		add_character_modifier = {
			modifier = efficient_tax_collection
			years = 20
		}
		
		every_held_title = {
			limit = { tier = tier_county }
			add_county_modifier = {
				modifier = reformed_taxation
				years = 20
			}
		}
	}
	
	option = {
		name = ottoman_economy.001.b
		
		add_trait = content
		add_stress = 25
		
		vassal_opinion = 10
	}
}

# Merchant Guild Establishment
ottoman_economy.002 = {
	type = character_event
	title = ottoman_economy.002.t
	desc = ottoman_economy.002.desc
	theme = stewardship
	
	left_portrait = {
		character = root
		animation = personality_rational
	}
	
	trigger = {
		is_ruler = yes
		culture = ottoman
		any_realm_county = {
			has_building = grand_bazaar_01
			development >= 15
		}
		NOT = { has_character_flag = merchant_guild_event }
		gold >= 400
	}
	
	immediate = {
		add_character_flag = merchant_guild_event
		random_realm_county = {
			limit = {
				has_building = grand_bazaar_01
				development >= 15
			}
			save_scope_as = trade_county
		}
	}
	
	option = {
		name = ottoman_economy.002.a
		
		add_gold = -400
		add_prestige = 150
		
		scope:trade_county = {
			add_county_modifier = {
				modifier = merchant_guild_established
				years = 50
			}
		}
		
		add_character_modifier = {
			modifier = trade_patron
			years = 25
		}
	}
	
	option = {
		name = ottoman_economy.002.b
		
		add_trait = greedy
		add_gold = 200
		
		different_culture_opinion = -10
	}
}

				modifier = agricultural_innovation
				years = 30
			}
		}
		
		every_realm_county = {
			limit = { terrain = farmlands }

		}

	option = {
		name = ottoman_economy.003.b

# Trade Route Expansion
ottoman_economy.004 = {
	type = character_event
	title = ottoman_economy.004.t
	desc = ottoman_economy.004.desc
	theme = diplomacy
	
	left_portrait = {
		character = root
		animation = personality_rational
	}
	
	trigger = {
		is_ruler = yes
		culture = ottoman
		diplomacy >= 14
		any_realm_county = {
			is_coastal_county = yes
		}
		NOT = { has_character_flag = trade_route_event }
		any_neighboring_ruler = {
			any_realm_county = {
				is_coastal_county = yes
			}
		}
	}
	
	immediate = {
		add_character_flag = trade_route_event
		random_neighboring_ruler = {
			limit = {
				any_realm_county = {
					is_coastal_county = yes
				}
			}
			save_scope_as = trade_partner
		}
	}
	
	option = {
		name = ottoman_economy.004.a
		
		add_gold = 500
		add_prestige = 200
		add_diplomacy_lifestyle_xp = 150
		
		add_character_modifier = {
			modifier = expanded_trade_routes
			years = 25
		}
		
		scope:trade_partner = {
			add_opinion = {
				target = root
				modifier = trade_partnership
				opinion = 30
			}
		}
	}
	
	option = {
		name = ottoman_economy.004.b
		
		add_trait = isolationist
		add_stress = 25
		
		same_culture_opinion = 10
		different_culture_opinion = -15
	}
}

# Craft Workshop Development
ottoman_economy.005 = {
	type = character_event
	title = ottoman_economy.005.t
	desc = ottoman_economy.005.desc
	theme = stewardship
	
	left_portrait = {
		character = root
		animation = personality_content
	}
	
	trigger = {
		is_ruler = yes
		culture = ottoman
		stewardship >= 12
		capital_county = {
			development >= 15
		}
		NOT = { has_character_flag = craft_workshop_event }
		any_courtier = {
			has_trait = master_craftsman
		}
	}
	
	immediate = {
		add_character_flag = craft_workshop_event
		random_courtier = {
			limit = { has_trait = master_craftsman }
			save_scope_as = master_craftsman
		}
	}
	
	option = {
		name = ottoman_economy.005.a
		
		add_gold = -300
		add_prestige = 100
		
		capital_county = {
			add_county_modifier = {
				modifier = craft_workshops
				years = 40
			}
		}
		
		scope:master_craftsman = {
			add_opinion = {
				target = root
				modifier = craft_patron
				opinion = 40
			}
		}
	}
	
	option = {
		name = ottoman_economy.005.b
		
		add_trait = content
		add_stress = -25
		
		court_opinion = 15
	}
}
